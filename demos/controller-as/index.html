Showing off the controller-as syntax. Notice how...

  • we attach the values to this instead of $scope inside the controller
  • we access the values via an alias in the HTML
{{ctrl.message}}
index.html
main.ts
<div ng-app="helloworld">
  <div ng-controller="helloworld-controller as ctrl">
    {{ctrl.message}}
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('helloworld', []);

app.controller('helloworld-controller', function(){

  // when using the controller-as syntax, we attach values
  // to "this" instead of to "$scope"!
  this.message = 'Hello world!';
});
