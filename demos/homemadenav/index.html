import * as angular from 'angular';

let app = angular.module('navDemo', []);

import addAppComponent from './appcomponent';
addAppComponent(app);

import addHomeComponent from './homecomponent';
addHomeComponent(app);

import addAboutComponent from './aboutcomponent';
addAboutComponent(app);

export default function addAboutComponent(app){
  app.component('aboutcomponent', {
    template: `
      <h4>About us</h4>
      We are an absolutely AMAZING company OMG OMG.
    `
  });
}
export default function addAppComponent(app){
  app.component('appcomponent', {
    controller: function($scope){
      $scope.position = 'home';
      $scope.goto = function(where){
        $scope.position = where;
      }
    },
    template: `
      <h1>Super app!</h1>
      <nav>
        <ul>
          <li ng-class="{active: position === 'home'}"
              ng-click="goto('home')">Home</li>
          <li ng-class="{active: position === 'about'}"
              ng-click="goto('about')">About</li>
        </ul>
      </nav>
      <homecomponent ng-if="position === 'home'"></homecomponent>
      <aboutcomponent ng-if="position === 'about'"></aboutcomponent>
      <footer>
        &copy; me until forever
      </footer>
    `
  });
}
export default function addHomeComponent(app){
  app.component('homecomponent', {
    template: `
      <h4>Home</h4>
      What an awesome site, AMIRITE?! :D
    `
  });
}
