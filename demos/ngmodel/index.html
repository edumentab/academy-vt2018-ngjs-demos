Demonstrating ng-model. It sets up a two-way data binding between a form input and a scope variable

<input ng-model="fieldVal"/>

Now the input and the fieldVal variable will always have the same value.



Current value: {{magic}}
index.html
main.ts
<div ng-app="ngmodel" ng-cloak>
  <div ng-controller="ngmodelcontroller">
    <input ng-model="magic"><br/>
    <input ng-model="magic"><br/>
    Current value: {{magic}}<br/>
    <button ng-click="costume()">Become Batman!</button>
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('ngmodel', []);

app.controller('ngmodelcontroller', function($scope){
  $scope.magic = 'initial value';
  $scope.costume = function(){
    $scope.magic = 'Batman';
  };
});
