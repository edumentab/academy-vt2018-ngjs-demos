Demonstrating how we can prevent the "flash of curly braces" before an app is loaded, by using the ng-cloak class which AngularJS automatically removes!

{{message}}
index.html
main.ts
styles.css
<div ng-app="helloworld" ng-cloak>
  <div ng-controller="helloworldcontroller">
    {{message}}
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('helloworld', []);

app.controller('helloworldcontroller', function($scope){
  $scope.message = 'Hello world!';
});

[ng-cloak] {
  display: none;
}
