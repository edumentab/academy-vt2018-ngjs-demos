  <div ng-controller="watchController">
    <input ng-model="fieldContent" placeholder="type sth!">
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('watch', []);

app.controller('watchController', function($scope){
  $scope.fieldContent = "Change me!";

  // our callback will be called whenever
  // the given expression changes
  $scope.$watch("fieldContent", function(newVal, oldVal){
    console.log("BEFORE:", oldVal, "NOW:", newVal);
  });
});
