<div ng-app="ngifDemo" ng-cloak>
  <div ng-controller="ngifController">
    <label>
      <input type="checkbox" ng-model="showSpoiler">
      Show spoiler
    </label>
    <p ng-if="showSpoiler">
      The lord commander will return.
    </p>
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('ngifDemo', []);

app.controller('ngifController', function($scope){
  $scope.showSpoiler = false; // initial value, not really needed
});
