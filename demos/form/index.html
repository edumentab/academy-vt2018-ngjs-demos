Demonstrating a basic form, showing event handling, data reading and a bunch of directives.

Current submission: {{currentVal}}
index.html
main.ts
<div ng-app="formDemo" ng-cloak>
  <div ng-controller="formController">
    <input ng-model="fieldVal">
    <button ng-disabled="!fieldVal.length"
            ng-click="submitValue(fieldVal)">Submit</button>
    <p>Current submission: {{currentVal}}</p>
  </div>
</div>
import * as angular from 'angular';

let app = angular.module('formDemo', []);

app.controller('formController', function($scope){
  $scope.submitValue = function(newVal){
    $scope.currentVal = newVal;
    $scope.fieldVal = ""; // emptying the field
  };
});
